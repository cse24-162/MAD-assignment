package assignment.accomodationfinderapp

import com.google.firebase.auth.ktx.auth
import com.google.firebase.firestore.ktx.firestore
import com.google.firebase.ktx.Firebase
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.tasks.await

object UserSession {

    private val _userName = MutableStateFlow("")
    val userName: StateFlow<String> = _userName

    private val _userRole = MutableStateFlow("")
    val userRole: StateFlow<String> = _userRole

    private val _userId = MutableStateFlow("")
    val userId: StateFlow<String> = _userId

    private val _userEmail = MutableStateFlow("")
    val userEmail: StateFlow<String> = _userEmail

    // Call this after login — fetches user data from Firestore.
    // userId is set immediately from FirebaseAuth so screens that depend on it
    // (SavedScreen, favorites queries) don't miss the first DisposableEffect run.
    suspend fun loadUser() {
        val uid = Firebase.auth.currentUser?.uid ?: return
        _userId.value = uid  // set before any async work
        try {
            val doc = Firebase.firestore
                .collection("users")
                .document(uid)
                .get()
                .await()

            _userName.value = doc.getString("name") ?: "User"
            _userRole.value = doc.getString("role") ?: "Student"
            _userEmail.value = doc.getString("email") ?: ""
        } catch (e: Exception) {
            _userName.value = "User"
        }
    }

    // Call this on logout
    fun clear() {
        _userName.value = ""
        _userRole.value = ""
        _userId.value = ""
        _userEmail.value = ""
    }

    // Check if someone is already logged in
    fun isLoggedIn(): Boolean {
        return Firebase.auth.currentUser != null
    }
}
